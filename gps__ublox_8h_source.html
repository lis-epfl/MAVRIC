58 /*
59  * try to put a UBlox into binary mode. This is in two parts. First we
60  * send a PUBX asking the UBlox to receive NMEA and UBX, and send UBX,
61  * with a baudrate of 38400. Then we send a UBX message setting rate 1
62  * for the NAV_SOL message. The setup of NAV_SOL is to cope with
63  * configurations where all UBX binary message types are disabled.
64  * 0001: UBX
65  * 0002: NMEA
66  * 0003: NMEA + UBX
67  */
68 //#define UBLOX_SET_BINARY "$PUBX,41,1,0003,0001,38400,0*26\n\265\142\006\001\003\000\001\006\001\022\117"
69 
70 // changed by ndousse
71 #define UBLOX_SET_BINARY "$PUBX,41,1,0003,0001,38400,0*25\r\n"//\265\142\006\001\003\000\001\006\001\022\117"

73 /*
74 Structure of an UBlox binary message
75 
76 PREAMBLE1 PREAMBLE2 CLASS MSG_ID LENGTH1 LENGTH2 MESSAGE CHECKSUMA CHECKSUMB
77 1 byte 1 byte 1 byte 1 byte Lowest byte Highest byte LENGTH size 1 byte 1 byte
78 
79 The information is received in the Little endian format (least significant byte first).
80 */
81 
82 #define UBX_PREAMBLE1 0xb5
83 #define UBX_PREAMBLE2 0x62
84 
85 /*
86 Name Class Description
87 NAV 0x01 Navigation Results: Position, Speed, Time, Acc, Heading, DOP, SVs used
88 RXM 0x02 Receiver Manager Messages: Satellite Status, RTC Status
89 INF 0x04 Information Messages: Printf-Style Messages, with IDs such as Error, Warning, Notice
90 ACK 0x05 Ack/Nack Messages: as replies to CFG Input Messages
91 CFG 0x06 Configuration Input Messages: Set Dynamic Model, Set DOP Mask, Set Baud Rate, etc.
92 MON 0x0A Monitoring Messages: Comunication Status, CPU Load, Stack Usage, Task Status
93 AID 0x0B AssistNow Aiding Messages: Ephemeris, Almanac, other A-GPS data input
94 TIM 0x0D Timing Messages: Timepulse Output, Timemark Results
95 */
