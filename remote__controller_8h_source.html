134 static inline void remote_controller_get_motor_state(int8_t *motor_state)
135 {
136  if((remote_controller_get_thrust_from_remote() < -0.95f) && (remote_controller_get_yaw_from_remote() > 0.9f))
137  {
138  *motor_state = 1;
139  }
140  else if((remote_controller_get_thrust_from_remote() < -0.95f) && (remote_controller_get_yaw_from_remote() < -0.9f))
141  {
142  *motor_state = -1;
143  }
144  else
145  {
146  *motor_state = 0;
147  }
148 }

171 static inline void remote_controller_get_channel_mode(uint8_t* chan_switch)
172 {
173  if (spektrum_satellite_get_channel(RC_SAFETY) < 0)
174  {
175  *chan_switch |= 0x00;
176  }
177  else if(spektrum_satellite_get_channel(RC_SAFETY) > 0 && spektrum_satellite_get_channel(RC_ID_MODE) < 0)
178  {
179  *chan_switch |= 0x01;
180  }
181  else if (spektrum_satellite_get_channel(RC_SAFETY) > 0 && spektrum_satellite_get_channel(RC_ID_MODE) > 20)
182  {
183  *chan_switch |= 0x03;
184  }
185  else
186  {
187  *chan_switch |= 0x02;
188  }
189 }
