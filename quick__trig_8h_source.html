127 static inline float quick_trig_func(float x, const float func_x_min, const float func_x_max, float func_x_step, const float func_y[])
128 {
129  float y;
130  int32_t i;
131  if ( x <= func_x_min )
132  {
133  y = func_y[0];
134  }
135  else if ( x >= func_x_max )
136  {
137  y = func_y[INTERP_POINTS - 1];
138  }
139  else
140  {
141  i = (int32_t) ((x - func_x_min) / func_x_step);
142  y = maths_interpolate(x,
143  func_x_min + i * func_x_step,
144  func_x_min + (i + 1) * func_x_step,
145  func_y[i],
146  func_y[i+1]);
147  }
148  return y;
149 }
