261 {

264  float snorm= SQR(q.s) + SQR(q.v[0]) + SQR(q.v[1]) + SQR(q.v[2]);
265 
266  if (snorm >0.0000001f)
267  {
268  float norm = maths_fast_sqrt(snorm);
269  result.s = q.s / norm;
270  result.v[0] = q.v[0] / norm;
271  result.v[1] = q.v[1] / norm;
272  result.v[2] = q.v[2] / norm;
273  }
274  else
275  {
276  result.s = 1.0f;
277  result.v[0] = 0.0f;
